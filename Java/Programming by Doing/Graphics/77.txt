//1. The window that appears is 800px wide and 600px tall.
//2. The first two numbers inside de g.drawRect are the position where the rectangle will start, the third one is the width 
//   of the dectangle and the last one is the height of the rectangle.  
//3. The numbers inside the g.fillOval are the same as the rectangle.
//4. The numbers insige the drawString are the pixels where the string starts.
//5. When 2 objects overlaps the onle that is listed last in the code is the one on top.

import java.util.*;
import java.awt.*;
import javax.swing.JFrame;


public class prac extends Canvas{
	
	 public void paint( Graphics g )
	    {
	        g.setColor(Color.green);
	        g.drawRect(50,20,100,200);  // draw a rectangle
	        g.fillOval(160,20,100,200); // draw a filled-in oval
	        g.setColor(Color.blue);
	        g.fillRect(200,400,200,20); // a filled-in rectangle
	        g.drawOval(200,430,200,100);
	        g.setColor(Color.red);
	        g.fillRect(680, 400, 100, 130);
	        
	        g.setColor(Color.black);
	        g.drawString("Graphics are pretty neat.", 500, 100);
	        int x = getWidth() / 2;
	        int y = getHeight() / 2;
	        g.drawString("The first letter of this string is at (" + x + "," + y + ")", x, y);
	    }

	    public static void main( String[] args )
	    {
	        // You can change the title or size here if you want.
	        JFrame win = new JFrame("Graphics");
	        win.setSize(800,600);
	        win.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
	        prac canvas = new prac();
	        win.add( canvas );
	        win.setVisible(true);
	    }
	}
